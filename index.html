🛠️ קוד HTML בסיסי לדוגמה:

html
<!DOCTYPE html>
<html>
<head>
  <title>תחרות קניות חשבונית</title>
</head>
<body>
  <h2>יש לכם 100 ש"ח - קנו מוצרים בלי לעבור את התקציב!</h2>
  <form id="shopping-form">
    <label><input type="checkbox" value="6.5"> חלב - 6.5 ש"ח</label><br>
    <label><input type="checkbox" value="5.9"> לחם - 5.9 ש"ח</label><br>
    <label><input type="checkbox" value="24.3"> עוגה - 24.3 ש"ח</label><br>
    <label><input type="checkbox" value="12"> קפה - 12 ש"ח</label><br>
    <label><input type="checkbox" value="8.2"> גבינה - 8.2 ש"ח</label><br><br>
    
    <button type="button" onclick="calculate()">חשב סכום</button>
  </form>

  <h3 id="result"></h3>

  <script>
    function calculate() {
      let checkboxes = document.querySelectorAll('input[type="checkbox"]:checked');
      let total = 0;
      checkboxes.forEach((item) => {
        total += parseFloat(item.value);
      });

      let budget = 100;
      let result = document.getElementById("result");

      if (total > budget) {
        result.innerHTML = "חרגת מהתקציב! (" + total.toFixed(2) + " ש"ח)";
      } else if (total === budget) {
        result.innerHTML = "מושלם! ניצלת את כל התקציב בדיוק!";
      } else {
        result.innerHTML = "הסכום שלך: " + total.toFixed(2) + " ש"ח | נשארו לך " + (budget - total).toFixed(2) + " ש"ח";
      }
    }
  </script>
</body>
</html>
